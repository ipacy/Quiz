import React, {useEffect, useState} from 'react';
import './Dashboard.css'
import Table from '@material-ui/core/Table';
import TableBody from '@material-ui/core/TableBody';
import TableCell from '@material-ui/core/TableCell';
import TableRow from '@material-ui/core/TableRow';
import {Link} from "react-router-dom";
import {FontIcon} from "office-ui-fabric-react";
import {Container} from "react-bootstrap";
import GlobalDb from "../../../../DBManager/db/GlobalDb";

/**
 * Component for Dashboard UI
 * This component shows analysis and progress
 * @component
 */
const DashboardUI = () => {
    const [myExams] = useState(0);
    const [exams, setExams] = useState(0);
    const [questions, SetQuestions] = useState(0);

    /**
     * @function
     */
    const handleGetCount = () => {
        GlobalDb.getDashBoard().then(response => {
            setExams(response.data['allExams'])
            SetQuestions(response.data['questions'])
        }).catch(e => {

        })

    }
    /**
     * @function
     * useEffect for Count API
     */
    useEffect(() => {
        handleGetCount();
    }, []);

    return (
        <Container className='DashboardTable'>
            <section>
                <Table>
                    <TableBody>
                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link className='THLink' to='/tutor_exams'>
                                    <FontIcon iconName="TestPlan"/>All Exams</Link>
                            </TableCell>
                            <TableCell align="right"><Link className='THLink'
                                                           to='/tutor_exams'>{exams}</Link></TableCell>
                        </TableRow>

                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link to='' className='THLink'> <FontIcon
                                    iconName="Questionnaire"/> Questions</Link>
                            </TableCell>
                            <TableCell align="right"><Link to='' className='THLink'>{questions}</Link></TableCell>
                        </TableRow>
                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link to='' className='THLink'>
                                    <FontIcon
                                        iconName="QuestionnaireMirrored"/> Exams Taken </Link> </TableCell>
                            <TableCell align="right"><Link to='' className='THLink'>6</Link></TableCell>
                        </TableRow>
                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link to='' className='THLink'> <FontIcon
                                    iconName="ModelingView"/> Categories</Link></TableCell>
                            <TableCell align="right"><Link to='' className='THLink'>6</Link></TableCell>
                        </TableRow>
                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link to='' className='THLink'> <FontIcon
                                    iconName="Certificate"/> Certificates</Link> </TableCell>
                            <TableCell align="right"><Link to='' className='THLink'>6</Link></TableCell>
                        </TableRow>
                    </TableBody>
                </Table>

                <Table>
                    <TableBody>
                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link to='' className='THLink'><FontIcon iconName="Group"/> Groups</Link>
                            </TableCell>
                            <TableCell align="right"><Link to='' className='THLink'>2</Link></TableCell>
                        </TableRow>
                        <TableRow>
                            <TableCell component="th" scope="row">
                                <Link to='' className='THLink'> <FontIcon iconName="ShowResults"/> Results</Link>
                            </TableCell>
                            <TableCell align="right"><Link to='' className='THLink'>31</Link></TableCell>
                        </TableRow>
                    </TableBody>
                </Table>
            </section>
        </Container>
    );
};

export default DashboardUI;
