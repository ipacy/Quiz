import DbManager from "../DbManager";


/**
 * Component Question DB Manager
 *
 * @component
 */
class QuestionDbClass {

    /**
     * @param   {string} examId  Exam Id
     * @param   {Object} question Question Data
     * @function
     * add questions
     */
    addQuestion = function (question, examId) {
        return DbManager.callServer("POST", `/v1/api/Question?examId=${examId}`, question);
    }

    /**
     * @param   {string} examId  Exam Id
     * @function
     * Get Questions by exam
     */
    getQuestionsByExam = function (examId) {
        return DbManager.callServer("GET", `/v1/api/Question/GetQuestionsByExam?examId=${examId}`);
    };

    /**
     * @param   {string} examId  Exam Id
     * @function
     * Get Questions by exam for users
     */
    getQuestionsByExamforUser = function (examId) {
        return DbManager.callServer("GET", `/v1/api/Question/GetQuestionsByExamforUser?examId=${examId}`);
    };

    /**
     * @param   {string} examId  Exam Id
     * @function
     * Get Questions by Id
     */
    getQuestionsById = function (examId) {
        return DbManager.callServer("GET", `/v1/api/Question/GetQuestionsById?examId=${examId}`);
    };

    /**
     * @param   {string} questionId  Question Id
     * @function
     * Get Questions by question Id
     */
    getOptionByQuestion = function (questionId) {
        return DbManager.callServer("GET", `/v1/api/Question/GetOptionsByQuestion?examId=${questionId}`);
    };

    getQuestions = function () {
        return DbManager.callServer("GET", '/v1/api/Question');
    };

    updateQuestion = function (question) {
        return DbManager.callServer("PUT", `/v1/api/Question`, question);
    };

    deleteQuestion = function (sId) {
        return DbManager.callServer("DELETE", `/v1/api/Question${sId}`);
    };
}

const QuestionDb = new QuestionDbClass();
export default QuestionDb;
